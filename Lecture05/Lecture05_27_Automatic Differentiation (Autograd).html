PyTorch Example
import torch # Create tensors with gradient tracking x = torch.tensor([2.0], requires_grad=True) w = torch.tensor([0.5], requires_grad=True) b = torch.tensor([1.0], requires_grad=True) # Forward pass (build computation graph) y = torch.sigmoid(w * x + b) loss = (y - 1)**2 # Backward pass (compute gradients) loss.backward() # Access gradients print(w.grad) # âˆ‚loss/âˆ‚w print(b.grad) # âˆ‚loss/âˆ‚b
